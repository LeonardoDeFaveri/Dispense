\section{Proprietà dei logaritmi}
\begin{enumerate}
    \setcounter{enumii}{0}
    \hitem{log}{\makebox[9.3cm]{$\log_a a=1$\hfill}\emph{Proprietà fondamentale}}
    \hitem{log}{\makebox[9.3cm]{$\log_a 1=0$\hfill}\emph{Proprietà fondamentale}}
    \hitem{log}{\makebox[9.3cm]{$\log_a(b\cdot c)=\log_a(b)+\log_a(c)$\hfill}\emph{Teorema del prodotto}}
    \hitem{log}{\makebox[9.3cm]{$\log_a\left(\frac{b}{c}\right)=\log_a(b)-\log_a(c)$\hfill}\emph{Teorema del rapporto}}
    \hitem{log}{\makebox[9.3cm]{$\log_a(b^c)=c\cdot\log_a(b)$\hfill}\emph{Teorema della potenza}}
    \hitem{log}{\makebox[9.3cm]{$\log_{a^n}(b^m)=\frac{m\cdot\log_a(b)}{n}$\hfill}\emph{Potenza alla base e all'argomento}}
    \hitem{log}{\makebox[9.3cm]{$\log_{\frac{1}{a}}(b)=-\log_a(b)$\hfill}\emph{Base frazionaria}}
    \hitem{log}{\makebox[9.3cm]{$\log_a\left(\frac{1}{b}\right)=-\log_b(a)$\hfill}\emph{Argomento frazionario}}
    \hitem{log}{\makebox[9.3cm]{$\log_{\frac{1}{a}}\left(\frac{1}{b}\right)=\log_a(b)$\hfill}\emph{Base e argomento frazionario}}
    \hitem{log}{\makebox[9.3cm]{$\log_a(b)=\frac{1}{\log_b(a)}$\hfill}\emph{Commutazione base e argomento}}
    \hitem{log}{\makebox[9.3cm]{$\log_a(b)=\frac{\log_c(b)}{\log_c(a)}$\hfill}\emph{Cambio di base}}
    \hitem{log}{\makebox[9.3cm]{$a^{\log_b(c)}=c^{\log_b(a)}$\hfill}\emph{Scambio base-argomento}}
\end{enumerate}

\section{Serie matematiche convergenti}
\begin{enumerate}
    \setcounter{enumii}{0}
    \hitem{ser}{\makebox[9.3cm]{$\sum_{k=0}^{+\infty}k=\frac{k(k+1)}{2}$\hfill}\emph{Formula di Gauss}}
    \hitem{ser}{\makebox[9.3cm]{$\sum_{k=0}^nq^k=\frac{1-q^{n+1}}{1-q}=\frac{q^{n+1}-1}{q-1}\quad\forall\,q:|q|\geq1$\hfill}\emph{Serie geometrica finita}}
    \hitem{ser}{\makebox[9.3cm]{$\sum_{k=0}^nq^k=\frac{1}{1-q}\quad\forall\,q:|q|<1$\hfill}\emph{Serie geometrica finita}}
    \hitem{ser}{\makebox[9.3cm]{$\sum_{k=0}^{+\infty}q^k=\frac{1}{1-q}\quad\forall\,|q|<1$\hfill}\emph{Serie geometrica infinita decrescente}}
    \hitem{ser}{\makebox[9.3cm]{$\sum_{k=0}^{+\infty}kq^k=\frac{q}{(1-q)^2}\quad\forall\,|q|<1$\hfill}\emph{Serie geometrica infinita decrescente}}
    \hitem{ser}{\makebox[9.3cm]{$\sum_{k=1}^{+\infty}\frac{1}{k(k+1)}=
    \sum_{k=1}^{+\infty}\left(\frac{1}{k}-\frac{1}{k+1}\right)=1$\hfill}\emph{Serie di Mengoli}}
\end{enumerate}
